Follow Command 

1. sudo apt update && sudo apt upgrade -y
2. sudo apt install -y curl
3. curl -sSfL 'https://github.com/GaiaNet-AI/gaianet-node/releases/latest/download/install.sh' | bash
4. source /root/.bashrc
5. gaianet init
6. gaianet start
7. gaianet info
